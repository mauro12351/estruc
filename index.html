//Handryd Murillo Valencia //Arley Camacho #include #include #include using namespace std; struct nodo { int placa; int tipo; // 1 es para carro y 2 es para la moto }; stack parqueadero; int totalRecaudado = 0; unordered_map placaToTipo; // sirve para identificar que la clave y como el valor que esta almacenado son enteros void registrar(int tipo, int placa) { nodo vehiculo; vehiculo.placa = placa; vehiculo.tipo = tipo; parqueadero.push(vehiculo); placaToTipo[placa] = tipo; // para almacenar la asociación entre la placa y el tipo de vehículo if (tipo == 1) totalRecaudado += 1000; // Valor del carro else if (tipo == 2) totalRecaudado += 500; // Valor de la moto cout << "Vehículo registrado correctamente." << endl; } void mostrar() { if (parqueadero.empty()) { cout << "La pila está vacía." << endl; return; } stack temp = parqueadero; cout << "Estado del parqueadero (LIFO):" << endl; cout << "---------------------------------" << endl; while (!temp.empty()) { nodo vehiculo = temp.top(); cout << "Placa: " << vehiculo.placa << ", Tipo: "; if (vehiculo.tipo == 1) cout << "Carro"; else if (vehiculo.tipo == 2) cout << "Moto"; cout << endl; temp.pop(); } } void eliminar(int placa) { if (parqueadero.empty()) { cout << "El parqueadero está vacío, no hay vehículos para retirar." << endl; return; } stack tempStack; while (!parqueadero.empty() && parqueadero.top().placa != placa) { int tipo = parqueadero.top().tipo; if (tipo == 1) totalRecaudado += 100; // este es el costo adicional por cada movimiento de los carros else if (tipo == 2) totalRecaudado += 50; // este es el costo adicional por cada movimiento de las motos tempStack.push(parqueadero.top()); parqueadero.pop(); } if (!parqueadero.empty()) { cout << "Vehículo con placa " << placa << " retirado correctamente." << endl; parqueadero.pop(); } else { cout << "No se encontró un vehículo con ese numero de placa " << placa << "." << endl; } while (!tempStack.empty()) { parqueadero.push(tempStack.top()); tempStack.pop(); } } int main() { int opcion; do { cout << "\nMenú:Paqueadero Para Vehiculos"; cout <<"\n1. Registrar vehículo\n2. Mostrar parqueadero\n3. Retirar vehículo\n4. Calcular valor recaudado\n5. Salir\n"; cout << "---------------------------------" << endl; cout << "Selecciona una opción: "; cin >> opcion; cout << "---------------------------------" << endl; switch (opcion) { case 1: { int tipo, placa; cout << "Seleccione el tipo de vehículo (1 para carro, 2 para moto): "; cin >> tipo; cout << "---------------------------------" << endl; if (tipo != 1 && tipo != 2) { cout << "Tipo de vehículo no válido." << endl; break; } cout << "Ingrese el número de la placa del vehículo: "; cin >> placa; cout << "---------------------------------" << endl; registrar(tipo, placa); break; } case 2: mostrar(); cout << "---------------------------------" << endl; break; case 3: { int placa; cout << "Ingrese el número de la placa del vehículo que desee retirar: "; cin >> placa; cout << "---------------------------------" << endl; eliminar(placa); break; } case 4: cout << "Total del recaudado hasta el momento: $" << totalRecaudado << endl; cout << "---------------------------------" << endl; break; case 5: cout << "Cerrando el programa para parqueadero de vehiculos. ¡Hasta luego!" << endl; cout << "---------------------------------" << endl; cout << "-----Futuros Programadores Arley Camacho y Murillo Handryd ;) -----" << endl; break; default: cout << "Opción inválida. Inténtalo de nuevo." << endl; } } while (opcion != 5); return 0; }
